    simgr.explore(find=lambda s: b"GRUB_ERR" in s.posix.dumps(1))
    if simgr.found":
        found_state = simgr.found[0]
        input_to_crash = found_state.solver.eval(sym_arg, cast_to=bytes)

        print("No vulnerabilities found with the given input size.")
if __name__ == "__main__":

import angr
import claripy
def main():
    project = angr.Project('./main_angr', auto_load_libs=False)
    stdin_size = 100
    sym_arg = claripy.BVS('sym_arg', stdin_size*8)
    state = project.factory.entry_state(args=['./main_angr'], stdin=sym_arg)
    simgr = project.factory.simulation_manager(state)
    def check stderr(state):
        stderr_output = state.posix.dumps(2)
        return len(stderr_output) > 0
    simgr.explore(find=check_stderr)
    if simgr.found":
        found_state = simgr.found[0]
        input_to_error = found_state.solver.eval(sym_arg, cast_to=bytes)
        print(f"Found input that leads to stderr output: {input_to_error}")
    else:
        print("No issues found with the given input size.")
if __name__ == "__main__":
    main()
